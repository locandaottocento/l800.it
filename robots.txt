  // ── COOKIE CONSENT ──
  (function() {
    var consent = localStorage.getItem('l800_cookie_consent');
    var banner  = document.getElementById('cookie-banner');
    if (consent === null) {
      // Prima visita: mostra banner dopo 800ms
      setTimeout(function() {
        if (banner) banner.classList.add('visible');
      }, 800);
    } else {
      // Consenso già espresso: nascondi definitivamente
      if (banner) banner.style.display = 'none';
      if (consent === 'true') enableThirdParty();
    }
  })();

  function setCookieConsent(accepted) {
    localStorage.setItem('l800_cookie_consent', accepted ? 'true' : 'false');
    var banner = document.getElementById('cookie-banner');
    if (banner) { banner.style.display = 'none'; }
    if (accepted) enableThirdParty();
  }

  function enableThirdParty() {
    // Attiva Google Maps iframe
    document.querySelectorAll('[data-src-consent]').forEach(function(el) {
      el.src = el.getAttribute('data-src-consent');
    });
  }

  // Funzione per riaprire banner dalla cookie policy
  window.riapriBannerCookie = function() {
    localStorage.removeItem('l800_cookie_consent');
    var banner = document.getElementById('cookie-banner');
    if (banner) banner.classList.add('visible');
  };
</script>
